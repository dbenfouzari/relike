import { Canvas, Meta, Story } from '@storybook/addon-docs';
import Brightness from "../brightness";
import Colors from './colors';
import MaterialColor from "../material-color";

export const ColorComponent = ({ color, name }) => (
  <div style={{
    width: 200,
    height: 100,
    display: 'grid',
    placeItems: 'center',
    backgroundColor: color.toRGBA(),
  }}>
    <span
      style={{ color: color.estimateBrightness() === Brightness.light ? Colors.black.toRGB() : Colors.white.toRGB() }}>{name}</span>
  </div>
)

export const PaletteComponent = ({ colorKey }) => {
  return (
    <div style={{ display: "flex" }}>
      {[50, 100, 200, 300, 400, 500, 600, 700, 800, 900].map(shade => (
          <ColorComponent key={`${colorKey}--${shade}`} color={Colors[colorKey][shade]} name={shade} />
      ))}
    </div>
  )
}

export const AccentPaletteComponent = ({ colorKey }) => {
  return (
    <div style={{ display: "flex" }}>
      {[100, 200, 400, 700].map(shade => (
          <ColorComponent key={`${colorKey}--${shade}`} color={Colors[colorKey][shade]} name={shade} />
      ))}
    </div>
  )
}

# Colors

You can use any Material Color.
All [Colors](?path=/docs/atoms-colors--page) render a [Color](?path=/docs/atoms-color--page), so you can manipulate them freely.

```tsx
import Colors from './colors';

const green = Colors.green[500];

return <div style={{ backgroundColor: green.toRGBA() }} />
```

<Canvas>
  <div style={{ width: 50, height: 50, backgroundColor: Colors.green[500].toRGBA() }} />
</Canvas>

## Usage

All colors can be used as `Colors.green[500]` or `Colors.green`. In fact, when you just call a color,
it calls the primary shade that is 500. So `Colors.green` acts like `Colors.green[500]`.

### Example

<Canvas>
  <div style={{ height: 50, backgroundColor: Colors.green.toRGBA() }}>Colors.green</div>
  <div style={{ height: 50, backgroundColor: Colors.green[500].toRGBA() }}>Colors.green[500]</div>
</Canvas>
